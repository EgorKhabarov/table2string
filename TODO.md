## v_align

Setting up vertical alignment
```pycon
>>> from table2string import Table, Themes
>>> table = [
...     ("11111", "22222"),
...     ("33333", "44444"),
... ]
>>> kwargs = {
...     "align": "^",
...     "theme": Themes.thin,
...     "max_width": (9,),
...     "max_height": 3,
...     "maximize_height": True,
... }
>>> Table(table).print(**kwargs)
┌───────────┬───────────┐
│   11111   │   22222   │
│           │           │
│           │           │
├───────────┼───────────┤
│   33333   │   44444   │
│           │           │
│           │           │
└───────────┴───────────┘
>>> # v_align in ("^", "-", "_")
>>> Table(table).print(**kwargs, v_align="-")  # doctest: +SKIP
┌───────────┬───────────┐
│           │           │
│   11111   │   22222   │
│           │           │
├───────────┼───────────┤
│           │           │
│   33333   │   44444   │
│           │           │
└───────────┴───────────┘

```

# Border Name

```pycon
>>> from table2string import Table, Themes
>>> table = [
...     ("red", "255,   0,   0"),
...     ("blue", "  0,   0, 255"),
...     ("Gray-brown-raspberry", "164,  69,  79"),
... ]
>>> kwargs = {
...     "theme": Themes.thin,
... }
>>> Table(table).print(**kwargs)
┌──────────────────────┬───────────────┐
│ red                  │ 255,   0,   0 │
├──────────────────────┼───────────────┤
│ blue                 │   0,   0, 255 │
├──────────────────────┼───────────────┤
│ Gray-brown-raspberry │ 164,  69,  79 │
└──────────────────────┴───────────────┘
>>> Table(table, name="Table Name").print(**kwargs)
┌──────────────────────────────────────┐
│              Table Name              │
├──────────────────────┬───────────────┤
│ red                  │ 255,   0,   0 │
├──────────────────────┼───────────────┤
│ blue                 │   0,   0, 255 │
├──────────────────────┼───────────────┤
│ Gray-brown-raspberry │ 164,  69,  79 │
└──────────────────────┴───────────────┘
>>> Table(table, border_name="Border Name").print(**kwargs)  # doctest: +SKIP
┌─Border─Name──────────┬───────────────┐
│ red                  │ 255,   0,   0 │
├──────────────────────┼───────────────┤
│ blue                 │   0,   0, 255 │
├──────────────────────┼───────────────┤
│ Gray-brown-raspberry │ 164,  69,  79 │
└──────────────────────┴───────────────┘
>>> Table(table, name="Table Name", border_name="Border Name").print(**kwargs)  # doctest: +SKIP
┌─Border─Name──────────────────────────┐    ┌──────────────────────────────────────┐
│              Table Name              │    │              Table Name              │
├──────────────────────┬───────────────┤    ├─Border─Name──────────┬───────────────┤
│ red                  │ 255,   0,   0 │    │ red                  │ 255,   0,   0 │
├──────────────────────┼───────────────┤ OR ├──────────────────────┼───────────────┤
│ blue                 │   0,   0, 255 │    │ blue                 │   0,   0, 255 │
├──────────────────────┼───────────────┤    ├──────────────────────┼───────────────┤
│ Gray-brown-raspberry │ 164,  69,  79 │    │ Gray-brown-raspberry │ 164,  69,  79 │
└──────────────────────┴───────────────┘    └──────────────────────┴───────────────┘
```
